DROP TABLE Directors;
DROP TABLE Actors;
DROP TABLE Movies;
DROP TABLE Payments;
DROP TABLE UserProfile;

DROP VIEW RECOMM;

CREATE TABLE UserProfile (
      UserID     VARCHAR2(15) NOT NULL,
      UserFName  VARCHAR2(25) NOT NULL,
      UserLName  VARCHAR2(25) NOT NULL,
      UserPASS   VARCHAR2(15) NOT NULL,
      CreditCard VARCHAR2(16) NOT NULL,
      PRIMARY KEY (UserID)
      );


CREATE TABLE Payments (
      TransactionID   VARCHAR2(25) NOT NULL,
      ItemPurchased   VARCHAR2(50),
      PurchaseDate    VARCHAR2(10),
      PurchaseAmount  VARCHAR2(8),
	  GENRE   		  VARCHAR2(25),
      PRIMARY KEY (TransactionID)
      );


CREATE TABLE Movies (
       MovieID     VARCHAR2(25) NOT NULL,
       MovieName   VARCHAR2(25),
       MovieYear   VARCHAR2(10),
       Rating 	   VARCHAR2(25),
	   GENRE       VARCHAR2(25),
       PRIMARY KEY (MovieID)
);


CREATE TABLE Actors (
    ActorID      VARCHAR2(25) NOT NULL,
    ActorName    VARCHAR2(35) NOT NULL,
    ActorBirth   VARCHAR2(10) NOT NULL,
    ActorBio     VARCHAR2(350),
    PRIMARY KEY (ActorID)
);


CREATE TABLE Directors (
    DirectorID     VARCHAR2(25) NOT NULL,
    DirectorName   VARCHAR2(35) NOT NULL,
    DirectorBirth  VARCHAR2(10) NOT NULL,
    DirectorBio    VARCHAR2(350),
    PRIMARY KEY (DirectorID)
    );
	
CREATE VIEW RECOMM AS
SELECT DISTINCT MOVIENAME,MOVIES.GENRE
FROM MOVIES, PAYMENTS
WHERE MOVIES.GENRE = PAYMENTS.GENRE AND NOT MOVIENAME = PAYMENTS.ITEMPURCHASED;
